    <script>
        const questions = [
            {
                text: "¿Qué porcentaje de tus clientes te contactan fuera de horario o fines de semana?",
                options: [
                    { text: "Casi todos, creen que soy soporte 24/7", points: 0 },
                    { text: "Unos pocos 'favoritos' que abusan", points: 10 },
                    { text: "Nadie, tengo procesos y límites claros", points: 20 }
                ]
            },
            {
                text: "Si mañana duplicas tus precios... ¿qué pasaría?",
                options: [
                    { text: "Se irían todos corriendo", points: 0 },
                    { text: "Se irían algunos, pero sobreviviría", points: 10 },
                    { text: "Se quedarían los mejores porque valoran mi resultado", points: 20 }
                ]
            },
            {
                text: "¿Cuántos de tus clientes actuales son 'referidos' que llegaron pidiendo descuento?",
                options: [
                    { text: "La mayoría, dependo solo de referidos baratos", points: 0 },
                    { text: "Algunos, pero trato de filtrar", points: 10 },
                    { text: "Muy pocos, yo elijo con quién trabajar", points: 20 }
                ]
            },
            {
                text: "¿Qué sientes el domingo por la noche al pensar en tu agenda del lunes?",
                options: [
                    { text: "Ansiedad, mis clientes me agotan", points: 0 },
                    { text: "Resignación, es lo que hay para pagar cuentas", points: 10 },
                    { text: "Entusiasmo, trabajo con gente que respeto", points: 20 }
                ]
            },
            {
                text: "¿Te sientes con la libertad de 'despedir' a tu peor cliente hoy mismo?",
                options: [
                    { text: "Imposible, necesito ese flujo de caja", points: 0 },
                    { text: "Podría, pero me dolería el bolsillo", points: 10 },
                    { text: "Sí, tengo prospectos en cola esperando", points: 20 }
                ]
            }
        ];

        let currentQuestion = 0;
        let totalScore = 0;
        let selections = new Array(questions.length).fill(null);

        function startAudit() {
            document.getElementById('welcome-screen').classList.add('hidden');
            document.getElementById('question-screen').classList.remove('hidden');
            showQuestion();
        }

        function showQuestion() {
            const q = questions[currentQuestion];
            document.getElementById('question-text').innerText = q.text;
            document.getElementById('step-counter').innerText = `Pregunta ${currentQuestion + 1} de ${questions.length}`;
            
            const progress = ((currentQuestion + 1) / questions.length) * 100;
            document.getElementById('progress-fill').style.width = `${progress}%`;
            document.getElementById('step-percent').innerText = `${Math.round(progress)}%`;

            const container = document.getElementById('options-container');
            container.innerHTML = '';

            q.options.forEach((opt, idx) => {
                const div = document.createElement('div');
                div.className = `option-card p-5 rounded-2xl font-medium text-gray-700 ${selections[currentQuestion] === idx ? 'selected' : ''}`;
                div.innerText = opt.text;
                div.onclick = () => selectOption(idx, opt.points);
                container.appendChild(div);
            });

            // Navegación
            const prevBtn = document.getElementById('prev-btn');
            if (currentQuestion > 0) {
                prevBtn.classList.remove('opacity-0', 'pointer-events-none');
            } else {
                prevBtn.classList.add('opacity-0', 'pointer-events-none');
            }

            const nextBtn = document.getElementById('next-btn');
            if (selections[currentQuestion] !== null) {
                nextBtn.classList.remove('bg-gray-100', 'text-gray-400', 'cursor-not-allowed');
                nextBtn.classList.add('bg-brand', 'text-white', 'hover:brightness-110');
            } else {
                nextBtn.classList.add('bg-gray-100', 'text-gray-400', 'cursor-not-allowed');
                nextBtn.classList.remove('bg-brand', 'text-white', 'hover:brightness-110');
            }
        }

        function selectOption(idx, points) {
            selections[currentQuestion] = idx;
            showQuestion();
        }

        function nextQuestion() {
            if (selections[currentQuestion] === null) return;
            
            if (currentQuestion < questions.length - 1) {
                currentQuestion++;
                showQuestion();
            } else {
                showResults();
            }
        }

        function prevQuestion() {
            if (currentQuestion > 0) {
                currentQuestion--;
                showQuestion();
            }
        }

        function showResults() {
            document.getElementById('question-screen').classList.add('hidden');
            document.getElementById('result-screen').classList.remove('hidden');

            const totalMax = questions.length * 20;
            const userScore = selections.reduce((acc, selIdx, qIdx) => {
                return acc + questions[qIdx].options[selIdx].points;
            }, 0);

            const percent = (userScore / totalMax) * 100;
            document.getElementById('score-circle').innerText = `${Math.round(percent)}%`;

            let verdict = "";
            let tag = "";

            if (percent >= 80) {
                tag = "Negocio Saludable";
                verdict = "Tienes un sistema sólido. Eres el dueño de tu tiempo y tus clientes te respetan. Tu próximo reto es escalar sin perder esta calidad de vida.";
            } else if (percent >= 50) {
                tag = "Estable pero Estancado";
                verdict = "No estás en crisis, pero estás 'cómodo' en un lugar peligroso. Tienes clientes que te quitan energía y te impiden buscar a los que pagan lo que realmente vales.";
            } else {
                tag = "En Cuidados Intensivos";
                verdict = "No tienes un negocio, eres rehén de tus clientes. Si sigues así, te vas a quemar. Necesitas un sistema Profit Factor para limpiar tu cartera y atraer clientes de calidad urgentemente.";
            }

            document.getElementById('verdict-text').innerText = `"${verdict}"`;
            document.getElementById('result-tag').innerText = tag;
        }
    </script>
